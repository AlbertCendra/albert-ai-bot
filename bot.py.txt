import os
import requests
from telegram import Update
from telegram.ext import Application, MessageHandler, filters, ContextTypes

# Ambil dari environment variables (lebih aman)
TELEGRAM_TOKEN = os.getenv("TELEGRAM_TOKEN")
OPENROUTER_API_KEY = os.getenv("OPENROUTER_KEY")

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_text = update.message.text
    
    try:
        response = requests.post(
            url="https://openrouter.ai/api/v1/chat/completions",
            headers={"Authorization": f"Bearer {OPENROUTER_API_KEY}"},
            json={
                "model": "google/gemma-4-26b-a4b-it:free",
                "messages": [
                    {"role": "system", "content": "Kamu adalah asisten AI pribadi Albert, ahli Software QA dan Odoo."},
                    {"role": "user", "content": user_text}
                ]
            }
        )
        ai_reply = response.json()['choices'][0]['message']['content']
        await update.message.reply_text(ai_reply)
    except Exception as e:
        await update.message.reply_text("Waduh, ada gangguan koneksi ke otak AI. Coba lagi ya!")

if __name__ == "__main__":
    app = Application.builder().token(TELEGRAM_TOKEN).build()
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    print("Bot sedang berjalan...")
    app.run_polling()
